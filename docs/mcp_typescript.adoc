== Introducción

El Model Context Protocol (MCP) es un protocolo que permite a las aplicaciones proporcionar contexto a los LLMs de forma estandarizada. Este documento cubre el SDK oficial de TypeScript para crear servidores y clientes MCP.

=== Requisitos Previos

* Node.js 18+ o Bun
* Conocimientos básicos de TypeScript
* Familiaridad con async/await

=== Instalación

[source,bash]
----
# Para servidores
npm install @modelcontextprotocol/server @modelcontextprotocol/node zod

# Para clientes
npm install @modelcontextprotocol/client

# Para integración Express
npm install @modelcontextprotocol/express

# Todo junto
npm install @modelcontextprotocol/server @modelcontextprotocol/node @modelcontextprotocol/client @modelcontextprotocol/express zod
----

== Parte 1: Servidor MCP Básico

=== Servidor mínimo con stdio

El transporte stdio es el más simple y se usa principalmente con Claude Desktop.

[source,typescript]
----
// server-basic.ts
import { McpServer } from '@modelcontextprotocol/server';

// Crear el servidor
const server = new McpServer({
  name: 'demo-server',
  version: '1.0.0'
});

// Registrar tool simple
server.registerTool(
  'hello',
  {
    title: 'Hello',
    description: 'Saluda a una persona por su nombre',
    inputSchema: z.object({
      name: z.string().describe('El nombre de la persona')
    })
  },
  async ({ name }) => {
    return {
      content: [{
        type: 'text',
        text: `¡Hola, ${name}! Bienvenido al servidor MCP.`
      }]
    };
  }
);

// Registrar tool con validación
server.registerTool(
  'add',
  {
    title: 'Add Numbers',
    description: 'Suma dos números enteros',
    inputSchema: z.object({
      a: z.number().int().describe('Primer número'),
      b: z.number().int().describe('Segundo número')
    })
  },
  async ({ a, b }) => {
    return {
      content: [{
        type: 'text',
        text: JSON.stringify({ result: a + b })
      }],
      structuredContent: { result: a + b }
    };
  }
);

// Ejecutar servidor
async function main() {
  const transport = new StdioServerTransport();
  await server.connect(transport);
}

main().catch(console.error);
----

// server-tools-advanced.ts
import { McpServer } from '@modelcontextprotocol/server';
import type { CallToolResult } from '@modelcontextprotocol/server';
import * as z from 'zod/v4';

const server = new McpServer({
  name: 'tools-advanced',
  version: '1.0.0'
});

// Schemas con validación avanzada
const UserRoleSchema = z.enum(['user', 'admin', 'moderator']);

const UserCreateSchema = z.object({
  username: z.string()
    .min(3, 'Username debe tener al menos 3 caracteres')
    .max(20, 'Username no puede tener más de 20 caracteres')
    .regex(/^[a-zA-Z0-9_]+$/, 'Solo letras, números y guión bajo'),
  email: z.string().email('Email inválido'),
  age: z.number()
    .int()
    .min(18, 'Debe ser mayor de edad')
    .max(120, 'Edad no válida'),
  role: UserRoleSchema.default('user'),
  bio: z.string().max(500).optional()
});

// Output schema
const UserResponseSchema = z.object({
  id: z.string(),
  username: z.string(),
  email: z.string(),
  age: z.number(),
  role: z.string(),
  created_at: z.string()
});

// Almacenamiento en memoria
const usersDb = new Map<string, z.infer<typeof UserResponseSchema>>();

// Registrar tool
server.registerTool(
  'create_user',
  {
    title: 'Create User',
    description: 'Crea un nuevo usuario en el sistema',
    inputSchema: z.object({
      user: UserCreateSchema
    }),
    outputSchema: UserResponseSchema,
    annotations: {
      openWorldHint: false
    }
  },
  async ({ user }): Promise<CallToolResult> => {
    // Verificar username único
    for (const existingUser of usersDb.values()) {
      if (existingUser.username === user.username.toLowerCase()) {
        return {
          content: [{
            type: 'text',
            text: `Error: El username '${user.username}' ya está en uso`
          }],
          isError: true
        };
      }
    }

    // Crear usuario
    const userId = randomUUID().slice(0, 8);
    const now = new Date().toISOString();
    
    const newUser = {
      id: userId,
      username: user.username.toLowerCase(),
      email: user.email,
      age: user.age,
      role: user.role,
      created_at: now
    };

    usersDb.set(userId, newUser);

    return {
      content: [{
        type: 'text',
        text: JSON.stringify(newUser, null, 2)
      }],
      structuredContent: newUser
    };
  }
);

// Tool para obtener usuario
server.registerTool(
  'get_user',
  {
    title: 'Get User',
    description: 'Obtiene un usuario por su ID',
    inputSchema: z.object({
      user_id: z.string().describe('ID del usuario')
    }),
    outputSchema: UserResponseSchema
  },
  async ({ user_id }): Promise<CallToolResult> => {
    const user = usersDb.get(user_id);
    
    if (!user) {
      return {
        content: [{
          type: 'text',
          text: `Error: Usuario '${user_id}' no encontrado`
        }],
        isError: true
      };
    }

    return {
      content: [{
        type: 'text',
        text: JSON.stringify(user, null, 2)
      }],
      structuredContent: user
    };
  }
);

// Tool para listar usuarios con filtros
server.registerTool(
  'list_users',
  {
    title: 'List Users',
    description: 'Lista usuarios con filtros opcionales',
    inputSchema: z.object({
      role: UserRoleSchema.optional(),
      min_age: z.number().int().min(0).optional(),
      limit: z.number().int().min(1).max(100).default(10)
    })
  },
  async ({ role, min_age, limit }): Promise<CallToolResult> => {
    let results = Array.from(usersDb.values());

    if (role) {
      results = results.filter(u => u.role === role);
    }

    if (min_age !== undefined) {
      results = results.filter(u => u.age >= min_age);
    }

    results = results.slice(0, limit);

    return {
      content: [{
        type: 'text',
        text: JSON.stringify(results, null, 2)
      }],
      structuredContent: { users: results, count: results.length }
    };
  }
);

// Tool destructiva
server.registerTool(
  'delete_user',
  {
    title: 'Delete User',
    description: 'Elimina un usuario del sistema',
    inputSchema: z.object({
      user_id: z.string()
    }),
    annotations: {
      destructiveHint: true,
      idempotentHint: true
    }
  },
  async ({ user_id }): Promise<CallToolResult> => {
    const user = usersDb.get(user_id);
    
    if (!user) {
      return {
        content: [{
          type: 'text',
          text: `Error: Usuario '${user_id}' no encontrado`
        }],
        isError: true
      };
    }

    usersDb.delete(user_id);

    return {
      content: [{
        type: 'text',
        text: JSON.stringify({ deleted: true, user }, null, 2)
      }]
    };
  }
);

async function main() {
  const transport = new StdioServerTransport();
  await server.connect(transport);
}

main().catch(console.error);

=== Tool con progreso y logging

[source,typescript]
----
// server-progress.ts
import { McpServer } from '@modelcontextprotocol/server';
import type { CallToolResult } from '@modelcontextprotocol/server';
import * as z from 'zod/v4';

const server = new McpServer(
  { name: 'progress-server', version: '1.0.0' },
  { capabilities: { logging: {} } }
);

server.registerTool(
  'process_files',
  {
    title: 'Process Files',
    description: 'Procesa múltiples archivos con reporte de progreso',
    inputSchema: z.object({
      files: z.array(z.string()).describe('Lista de archivos a procesar'),
      delay: z.number().default(500).describe('Delay entre archivos (ms)')
    })
  },
  async ({ files, delay }, ctx): Promise<CallToolResult> => {
    const progressToken = ctx.mcpReq._meta?.progressToken;
    
    await ctx.mcpReq.log('info', `Iniciando procesamiento de ${files.length} archivos`);
    
    const processed: string[] = [];
    const errors: string[] = [];

    for (let i = 0; i < files.length; i++) {
      const filename = files[i];

      // Reportar progreso
      if (progressToken !== undefined) {
        await ctx.mcpReq.notify({
          method: 'notifications/progress',
          params: {
            progressToken,
            progress: i + 1,
            total: files.length,
            message: `Procesando archivo ${i + 1}/${files.length}: ${filename}`
          }
        });
      }

      await ctx.mcpReq.log('debug', `Iniciando: ${filename}`);

      // Simular procesamiento
      await new Promise(resolve => setTimeout(resolve, delay));

      // Simular error
      if (filename.endsWith('.error')) {
        await ctx.mcpReq.log('warning', `Error en archivo: ${filename}`);
        errors.push(filename);
      } else {
        await ctx.mcpReq.log('debug', `Completado: ${filename}`);
        processed.push(filename);
      }
    }

    await ctx.mcpReq.log(
      'info',
      `Procesamiento completado: ${processed.length} exitosos, ${errors.length} errores`
    );

    return {
      content: [{
        type: 'text',
        text: JSON.stringify({
          total: files.length,
          processed,
          errors,
          success_rate: processed.length / files.length
        }, null, 2)
      }]
    };
  }
);

server.registerTool(
  'long_calculation',
  {
    title: 'Long Calculation',
    description: 'Realiza un cálculo largo con actualizaciones',
    inputSchema: z.object({
      n: z.number().int().positive(),
      steps: z.number().int().min(1).max(100).default(10)
    })
  },
  async ({ n, steps }, ctx): Promise<CallToolResult> => {
    const progressToken = ctx.mcpReq._meta?.progressToken;
    const startTime = Date.now();
    
    await ctx.mcpReq.log('info', `Iniciando cálculo con n=${n}, steps=${steps}`);

    let result = 0;

    for (let i = 0; i < steps; i++) {
      result += n ** 2;

      if (progressToken !== undefined) {
        await ctx.mcpReq.notify({
          method: 'notifications/progress',
          params: {
            progressToken,
            progress: i + 1,
            total: steps,
            message: `Paso ${i + 1}/${steps} completado`
          }
        });
      }

      await new Promise(resolve => setTimeout(resolve, 200));
    }

    const elapsed = Date.now() - startTime;
    await ctx.mcpReq.log('info', `Cálculo completado en ${elapsed}ms`);

    return {
      content: [{
        type: 'text',
        text: JSON.stringify({
          input: n,
          steps,
          result,
          elapsed_ms: elapsed
        }, null, 2)
      }]
    };
  }
);

async function main() {
  const transport = new StdioServerTransport();
  await server.connect(transport);
}

main().catch(console.error);

// ============== Modelos ==============

type Task = z.infer<typeof TaskSchema>;

// ============== Almacenamiento ==============

const tasksDb = new Map<string, Task>();

// ============== Tools ==============

  'create_task',

// client-interactive.ts
import * as readline from 'readline';
import { Client, StreamableHTTPClientTransport } from '@modelcontextprotocol/client';
import type { Tool, Resource } from '@modelcontextprotocol/client';

class MCPInteractiveClient {
  private client: Client;
  private rl: readline.Interface;

  constructor() {
    this.client = new Client({ name: 'interactive-client', version: '1.0.0' });
    this.rl = readline.createInterface({
      input: process.stdin,
      output: process.stdout
    });
  }

  async connect(url: string) {
    const transport = new StreamableHTTPClientTransport(new URL(url));
    await this.client.connect(transport);
    console.log('✓ Conectado al servidor');

    const instructions = this.client.getInstructions();
    if (instructions) {
      console.log(`\n📖 ${instructions}\n`);
    }
  }

  async disconnect() {
    await this.client.close();
    this.rl.close();
  }

  private async prompt(question: string): Promise<string> {
    return new Promise((resolve) => {
      this.rl.question(question, resolve);
    });
  }

  async listTools() {
    const result = await this.client.listTools();
    console.log('\n📋 Tools disponibles:');
    for (const tool of result.tools) {
      console.log(`  • ${tool.name}`);
      console.log(`    ${tool.description}`);
    }
    return result.tools;
  }

  async listResources() {
    const result = await this.client.listResources();
    console.log('\n📁 Resources:');
    for (const resource of result.resources) {
      console.log(`  • ${resource.uri}: ${resource.name}`);
    }
    return result.resources;
  }

  async callTool(name: string, args: Record<string, unknown>) {
    console.log(`\n🔧 Ejecutando: ${name}`);
    console.log(`   Argumentos:`, args);

    const result = await this.client.callTool({ name, arguments: args });

    if (result.isError) {
      console.log(`   ❌ Error:`, result.content[0]);
    } else {
      console.log(`   ✓ Resultado:`, result.content[0]);
    }

    return result;
  }

  async readResource(uri: string) {
    console.log(`\n📖 Leyendo: ${uri}`);
    const result = await this.client.readResource({ uri });
    console.log(`   Contenido:`, result.contents[0]);
    return result;
  }

  async runREPL() {
    console.log('\n' + '='.repeat(50));
    console.log('MCP Client REPL');
    console.log('Comandos: tools, resources, call <name> <json>, read <uri>, quit');
    console.log('='.repeat(50));

    while (true) {
      const cmd = await this.prompt('\n> ');

      if (!cmd.trim()) continue;

      const parts = cmd.trim().split(/\s+/);
      const command = parts[0];

      try {
        switch (command) {
          case 'quit':
          case 'exit':
            return;

          case 'tools':
            await this.listTools();
            break;

          case 'resources':
            await this.listResources();
            break;

          case 'call': {
            const toolName = parts[1];
            const argsJson = parts.slice(2).join(' ') || '{}';
            const args = JSON.parse(argsJson);
            await this.callTool(toolName, args);
            break;
          }

          case 'read': {
            const uri = parts[1];
            await this.readResource(uri);
            break;
          }

          default:
            console.log(`Comando desconocido: ${command}`);
        }
      } catch (error) {
        console.log('Error:', error);
      }
    }
  }
}

async function main() {
  const client = new MCPInteractiveClient();

  try {
    await client.connect('http://localhost:3000/mcp');
    await client.runREPL();
  } finally {
    await client.disconnect();
  }

  console.log('\n¡Hasta pronto!');
}

  }

server.registerTool(
  'list_books',
  {
    title: 'List Books',
    description: 'Lista libros con filtros',
    inputSchema: z.object({
      genre: z.string().optional(),
      available_only: z.boolean().default(false),
      limit: z.number().int().min(1).max(100).default(20)
    })
  },
  async ({ genre, available_only, limit }): Promise<CallToolResult> => {
    let results = Array.from(booksDb.values());

    if (genre) {
      results = results.filter(b => b.genre.toLowerCase() === genre.toLowerCase());
    }

    if (available_only) {
      results = results.filter(b => b.available);
    }

    results = results.slice(0, limit);

    return {
      content: [{ type: 'text', text: JSON.stringify(results, null, 2) }],
      structuredContent: { books: results, count: results.length }
    };
  }
);

  'search_books',
  {
    title: 'Search Books',
    description: 'Busca libros por título o autor',
    inputSchema: z.object({ query: z.string() })

  async ({ query }): Promise<CallToolResult> => {
    const queryLower = query.toLowerCase();
    
    const results = Array.from(booksDb.values()).filter(
      b => b.title.toLowerCase().includes(queryLower) ||

    };
  }

server.registerTool(
  'borrow_book',
  {
    title: 'Borrow Book',
    description: 'Presta un libro a un miembro',
    inputSchema: z.object({
      book_id: z.string(),
      member_id: z.string()
    })
  },
  async ({ book_id, member_id }): Promise<CallToolResult> => {
    const book = booksDb.get(book_id);
    const member = membersDb.get(member_id);

    if (!book) {
      return {
        content: [{ type: 'text', text: `Libro '${book_id}' no encontrado` }],
        isError: true
      };
    }

    if (!member) {
      return {
        content: [{ type: 'text', text: `Miembro '${member_id}' no encontrado` }],
        isError: true
      };
    }

    if (!book.available) {
      return {
        content: [{ type: 'text', text: `El libro '${book.title}' no está disponible` }],
        isError: true
      };
    }

    if (member.borrowed_books.length >= 5) {
      return {
        content: [{ type: 'text', text: 'El miembro ya tiene el máximo de libros (5)' }],
        isError: true
      };
    }

    // Realizar préstamo
    book.available = false;
    book.borrowed_by = member_id;
    book.borrowed_at = new Date().toISOString();
    member.borrowed_books.push(book_id);

    return {
      content: [{
        type: 'text',
        text: JSON.stringify({
          success: true,
          book: { id: book.id, title: book.title },
          member: member.name,
          borrowed_at: book.borrowed_at
        }, null, 2)
      }]
    };
  }
);

server.registerTool(
  'return_book',
  {
    title: 'Return Book',
    description: 'Devuelve un libro prestado',
    inputSchema: z.object({ book_id: z.string() })
  },
  async ({ book_id }): Promise<CallToolResult> => {
    const book = booksDb.get(book_id);

    if (!book) {
      return {
        content: [{ type: 'text', text: `Libro '${book_id}' no encontrado` }],
        isError: true
      };
    }

    if (book.available) {
      return {
        content: [{ type: 'text', text: 'Este libro no está prestado' }],
        isError: true
      };
    }

    const memberId = book.borrowed_by;

    if (memberId) {
      const member = membersDb.get(memberId);
      if (member) {
        member.borrowed_books = member.borrowed_books.filter(id => id !== book_id);
      }
    }

    book.available = true;
    book.borrowed_by = null;
    book.borrowed_at = null;

    return {
      content: [{
        type: 'text',
        text: JSON.stringify({
          success: true,
          book: { id: book.id, title: book.title },
          returned_by: memberId
        }, null, 2)
      }]
    };
  }
);

server.registerResource(
  'library-stats',
  'library://stats',
  {
    title: 'Library Statistics',
    description: 'Estadísticas de la biblioteca',
    mimeType: 'application/json'
  },
  async (uri) => {
    const books = Array.from(booksDb.values());
    
    const genres: Record<string, number> = {};
    for (const book of books) {
      genres[book.genre] = (genres[book.genre] || 0) + 1;
    }

    const stats = {
      total_books: books.length,
      available_books: books.filter(b => b.available).length,
      borrowed_books: books.filter(b => !b.available).length,
      total_members: membersDb.size,
      genres
    };

    return {
      contents: [{
        uri: uri.href,
        mimeType: 'application/json',
        text: JSON.stringify(stats, null, 2)
      }]
    };
  }
);

|===
| Método | Uso

| Conecta al servidor

| `client.listTools()`
| Lista herramientas

| `client.callTool({ name, arguments })`
| Invoca una herramienta

| `client.listResources()`
| Lista resources

| `client.readResource({ uri })`
| Lee un resource

| `client.listPrompts()`
| Lista prompts

| `client.getPrompt({ name, arguments })`
| Obtiene un prompt

| Cierra la conexión
