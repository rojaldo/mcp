Introducción a MCP y Clientes

¿Qué es MCP?

Model Context Protocol (MCP) es un protocolo abierto que estandariza cómo los modelos de lenguaje (LLMs) se conectan con fuentes de datos y herramientas externas. Proporciona una forma unificada de:

  • Exponer herramientas (tools) que los LLMs pueden invocar

  • Proporcionar recursos (resources) con datos contextuales

  • Ofrecer prompts predefinidos para tareas comunes

Arquitectura MCP

La arquitectura MCP sigue un modelo cliente-servidor:

┌─────────────────┐     ┌─────────────────┐     ┌─────────────────┐
│   Cliente MCP   │◄───►│    Transporte   │◄───►│  Servidor MCP   │
│   (FastMCP)     │     │   (stdio/HTTP)  │     │   (FastMCP)     │
└─────────────────┘     └─────────────────┘     └─────────────────┘
         │                                              │
         │                                              │
         ▼                                              ▼
┌─────────────────┐                           ┌─────────────────┐
│   Aplicación    │                           │   Herramientas  │
│   (LLM/Agente)  │                           │   Recursos       │
└─────────────────┘                           │   Prompts        │
                                              └─────────────────┘

Componentes Principales

Componente Descripción

Servidor MCP

Expone capacidades: tools, resources, prompts. Implementado con FastMCP usando decoradores.

Cliente MCP

Consume las capacidades del servidor. Puede ser una aplicación, un agente, o una CLI.

Transporte

Canal de comunicación. Opciones: stdio (más común), HTTP/SSE, WebSocket.

Tool

Función invocable que realiza una acción. Puede modificar estado, hacer llamadas API, etc.

Resource

Datos contextuales de solo lectura. Archivos, configuraciones, datos de BD.

Prompt

Plantilla de prompt predefinida. Ayuda a guiar al LLM en tareas específicas.

import json

mcp = FastMCP("mi-servidor", version="1.0.0")

@mcp.tool
def calcular_suma(
    a: Annotated[int, Field(description="Primer número", ge=0)],
    b: Annotated[int, Field(description="Segundo número", ge=0)]
) -> dict:
    """
    Calcula la suma de dos números.

    Args:
        a: Primer número (entero no negativo)
        b: Segundo número (entero no negativo)

    Returns:
        dict con el resultado de la operación
    """
    return {
        "operacion": "suma",
        "a": a,
        "b": b,
        "resultado": a + b
    }

@mcp.tool
def buscar_usuario(
    email: Annotated[str, Field(description="Email del usuario")]
) -> dict:
    """
    Busca un usuario por email.

    Args:
        email: Email del usuario a buscar

    Returns:
        dict con información del usuario o error si no existe
    """
    # Simulación de base de datos
    usuarios = {
        "juan@example.com": {"id": 1, "nombre": "Juan", "activo": True},
        "maria@example.com": {"id": 2, "nombre": "María", "activo": True},
    }

    if email in usuarios:
        return {"success": True, "usuario": usuarios[email]}
    return {"success": False, "error": "Usuario no encontrado"}

@mcp.tool
def crear_reporte(
    titulo: Annotated[str, Field(description="Título del reporte")],
    contenido: Annotated[str, Field(description="Contenido del reporte")],
    formato: Annotated[str, Field(description="Formato: markdown, html, pdf")] = "markdown"
) -> dict:
    """
    Crea un reporte con el contenido especificado.

    Args:
        titulo: Título del reporte
        contenido: Contenido del reporte
        formato: Formato de salida (markdown, html, pdf)

    Returns:
        dict con el reporte generado
    """
    reportes = {
        "markdown": f"# {titulo}\n\n{contenido}",
        "html": f"<h1>{titulo}</h1><p>{contenido}</p>",
        "pdf": f"[PDF] {titulo}: {contenido}"
    }

    return {
        "success": True,
        "formato": formato,
        "reporte": reportes.get(formato, reportes["markdown"])
    }

def prompt_generar_documentacion(
    funcion: str,

Clientes MCP: Casos de Uso por Tipo

Clientes que Utilizan TOOLS

Las herramientas (tools) son funciones que el cliente puede invocar para realizar acciones. Son ideales para operaciones que modifican estado, hacen llamadas a APIs externas, o realizan cálculos.

Caso de Uso 1: Automatización de Tareas

# client_tools_automation.py
"""
Cliente MCP para automatización de tareas usando tools.

Caso de uso: Automatizar operaciones de CRM como crear contactos,
enviar emails y registrar actividades.
"""
from fastmcp import Client
from fastmcp.client.stdio import StdioServerParameters
from fastmcp.client.transports import StdioTransport
import asyncio
from typing import List, Dict, Any

class CRMAutomationClient:
    """Cliente para automatizar operaciones de CRM."""

    def __init__(self, server_path: str):
        self.server_path = server_path
        self._client: Client = None

    async def conectar(self):
        """Conecta al servidor MCP."""
        server_params = StdioServerParameters(
            command="python",
            args=[self.server_path]
        )
        self._client = Client(StdioTransport(server_params))
        await self._client.connect()

        # Verificar herramientas disponibles
        tools = await self._client.list_tools()
        print(f"Conectado. {len(tools)} herramientas disponibles.")
        for tool in tools:
            print(f"  - {tool.name}: {tool.description}")

    async def crear_contacto(
        self,
        nombre: str,
        email: str,
        empresa: str = None,
        telefono: str = None
    ) -> Dict[str, Any]:
        """
        Crea un nuevo contacto en el CRM.

        Args:
            nombre: Nombre completo del contacto
            email: Email del contacto
            empresa: Empresa (opcional)
            telefono: Teléfono (opcional)

        Returns:
            dict con el contacto creado
        """
        args = {"nombre": nombre, "email": email}
        if empresa:
            args["empresa"] = empresa
        if telefono:
            args["telefono"] = telefono

        resultado = await self._client.call_tool("crear_contacto", args)
        return resultado

    async def enviar_email(
        self,
        destinatario: str,
        asunto: str,
        cuerpo: str,
        template: str = None
    ) -> Dict[str, Any]:
        """
        Envía un email usando una plantilla.

        Args:
            destinatario: Email del destinatario
            asunto: Asunto del email
            cuerpo: Cuerpo del mensaje
            template: Plantilla a usar (opcional)

        Returns:
            dict con el resultado del envío
        """
        args = {
            "destinatario": destinatario,
            "asunto": asunto,
            "cuerpo": cuerpo
        }
        if template:
            args["template"] = template

        resultado = await self._client.call_tool("enviar_email", args)
        return resultado

    async def registrar_actividad(
        self,
        contacto_id: int,
        tipo: str,
        descripcion: str,
        fecha: str = None
    ) -> Dict[str, Any]:
        """
        Registra una actividad para un contacto.

        Args:
            contacto_id: ID del contacto
            tipo: Tipo de actividad (llamada, email, reunion)
            descripcion: Descripción de la actividad
            fecha: Fecha de la actividad (opcional)

        Returns:
            dict con la actividad registrada
        """
        args = {
            "contacto_id": contacto_id,
            "tipo": tipo,
            "descripcion": descripcion
        }
        if fecha:
            args["fecha"] = fecha

        resultado = await self._client.call_tool("registrar_actividad", args)
        return resultado

    async def buscar_contactos(self, query: str) -> List[Dict[str, Any]]:
        """
        Busca contactos que coincidan con la query.

        Args:
            query: Término de búsqueda

        Returns:
            list de contactos encontrados
        """
        resultado = await self._client.call_tool(
            "buscar_contactos",
            {"query": query}
        )
        return resultado.get("contactos", [])

    async def crear_campana(
        self,
        nombre: str,
        descripcion: str,
        contactos: List[int]
    ) -> Dict[str, Any]:
        """
        Crea una campaña de marketing.

        Args:
            nombre: Nombre de la campaña
            descripcion: Descripción
            contactos: Lista de IDs de contactos

        Returns:
            dict con la campaña creada
        """
        resultado = await self._client.call_tool(
            "crear_campana",
            {
                "nombre": nombre,
                "descripcion": descripcion,
                "contactos": contactos
            }
        )
        return resultado

    async def ejecutar_campana(
        self,
        campana_id: int,
        plantilla_email: str
    ) -> Dict[str, Any]:
        """
        Ejecuta una campaña enviando emails a todos los contactos.

        Args:
            campana_id: ID de la campaña
            plantilla_email: Plantilla del email

        Returns:
            dict con estadísticas de la campaña
        """
        # Obtener campaña
        campana = await self._client.call_tool(
            "obtener_campana",
            {"id": campana_id}
        )

        # Obtener contactos de la campaña
        contactos = campana.get("contactos", [])

        # Enviar emails
        resultados = []
        for contacto_id in contactos:
            contacto = await self._client.call_tool(
                "obtener_contacto",
                {"id": contacto_id}
            )

            resultado = await self.enviar_email(
                destinatario=contacto["email"],
                asunto=campana["nombre"],
                cuerpo=campana["descripcion"],
                template=plantilla_email
            )
            resultados.append(resultado)

            # Registrar actividad
            await self.registrar_actividad(
                contacto_id=contacto_id,
                tipo="email",
                descripcion=f"Email enviado: {campana['nombre']}"
            )

        return {
            "campana_id": campana_id,
            "total_enviados": len(resultados),
            "exitosos": sum(1 for r in resultados if r.get("success")),
            "fallidos": sum(1 for r in resultados if not r.get("success"))
        }

    async def desconectar(self):
        """Desconecta del servidor."""
        if self._client:
            await self._client.disconnect()


# Ejemplo de uso
async def main():
    cliente = CRMAutomationClient("crm_server.py")

    try:
        await cliente.conectar()

        # Crear contacto
        contacto = await cliente.crear_contacto(
            nombre="Juan Pérez",
            email="juan@empresa.com",
            empresa="Tech Corp"
        )
        print(f"Contacto creado: {contacto}")

        # Enviar email
        email_result = await cliente.enviar_email(
            destinatario="juan@empresa.com",
            asunto="Bienvenido",
            cuerpo="Gracias por registrarte..."
        )
        print(f"Email enviado: {email_result}")

        # Registrar actividad
        actividad = await cliente.registrar_actividad(
            contacto_id=contacto["id"],
            tipo="email",
            descripcion="Email de bienvenida enviado"
        )
        print(f"Actividad registrada: {actividad}")

    finally:
        await cliente.desconectar()

asyncio.run(main())

# client_tools_processing.py
"""
Cliente MCP para procesamiento de datos usando tools.

Caso de uso: Pipeline de ETL (Extract, Transform, Load) que
procesa archivos de datos y los carga en una base de datos.
"""
from fastmcp import Client
from fastmcp.client.stdio import StdioServerParameters
from fastmcp.client.transports import StdioTransport
import asyncio
from typing import List, Dict, Any, Optional
from pathlib import Path

class ETLClient:
    """Cliente para pipeline de ETL."""

    def __init__(self, server_path: str):
        self.server_path = server_path
        self._client: Client = None

    async def conectar(self):
        """Conecta al servidor MCP."""
        server_params = StdioServerParameters(
            command="python",
            args=[self.server_path]
        )
        self._client = Client(StdioTransport(server_params))
        await self._client.connect()

    async def extraer_datos(
        self,
        fuente: str,
        formato: str = "csv",
        opciones: dict = None
    ) -> Dict[str, Any]:
        """
        Extrae datos de una fuente.

        Args:
            fuente: Ruta al archivo o conexión a BD
            formato: Formato de los datos (csv, json, excel)
            opciones: Opciones adicionales

        Returns:
            dict con los datos extraídos
        """
        args = {
            "fuente": fuente,
            "formato": formato
        }
        if opciones:
            args["opciones"] = opciones

        return await self._client.call_tool("extraer_datos", args)

    async def transformar_datos(
        self,
        datos: List[Dict],
        transformaciones: List[Dict[str, Any]]
    ) -> List[Dict]:
        """
        Aplica transformaciones a los datos.

        Args:
            datos: Lista de registros a transformar
            transformaciones: Lista de transformaciones a aplicar

        Returns:
            list con los datos transformados
        """
        return await self._client.call_tool(
            "transformar_datos",
            {
                "datos": datos,
                "transformaciones": transformaciones
            }
        )

    async def validar_datos(
        self,
        datos: List[Dict],
        reglas: List[Dict[str, Any]]
    ) -> Dict[str, Any]:
        """
        Valida datos según reglas.

        Args:
            datos: Lista de registros a validar
            reglas: Lista de reglas de validación

        Returns:
            dict con resultado de validación
        """
        return await self._client.call_tool(
            "validar_datos",
            {
                "datos": datos,
                "reglas": reglas
            }
        )

    async def cargar_datos(
        self,
        datos: List[Dict],
        destino: str,
        tabla: str,
        modo: str = "insert"
    ) -> Dict[str, Any]:
        """
        Carga datos en el destino.

        Args:
            datos: Lista de registros a cargar
            destino: Conexión a la base de datos
            tabla: Nombre de la tabla
            modo: Modo de carga (insert, upsert, replace)

        Returns:
            dict con estadísticas de carga
        """
        return await self._client.call_tool(
            "cargar_datos",
            {
                "datos": datos,
                "destino": destino,
                "tabla": tabla,
                "modo": modo
            }
        )

    async def ejecutar_pipeline(
        self,
        config: Dict[str, Any]
    ) -> Dict[str, Any]:
        """
        Ejecuta un pipeline completo de ETL.

        Args:
            config: Configuración del pipeline

        Returns:
            dict con resultados del pipeline
        """
        resultados = {
            "extraccion": None,
            "transformacion": None,
            "validacion": None,
            "carga": None,
            "estadisticas": {}
        }

        # Extraer
        print(f"Extrayendo datos de {config['fuente']}...")
        datos_extraidos = await self.extraer_datos(
            fuente=config["fuente"],
            formato=config.get("formato", "csv"),
            opciones=config.get("opciones_extraccion")
        )
        resultados["extraccion"] = datos_extraidos

        # Transformar
        print("Aplicando transformaciones...")
        datos_transformados = await self.transformar_datos(
            datos=datos_extraidos["datos"],
            transformaciones=config.get("transformaciones", [])
        )
        resultados["transformacion"] = datos_transformados

        # Validar
        print("Validando datos...")
        validacion = await self.validar_datos(
            datos=datos_transformados,
            reglas=config.get("reglas_validacion", [])
        )
        resultados["validacion"] = validacion

        if not validacion.get("valid", True):

"""
Cliente MCP para gestión de configuración usando resources.

Caso de uso: Aplicación que lee configuraciones dinámicamente

from fastmcp import Client
from fastmcp.client.stdio import StdioServerParameters
from fastmcp.client.transports import StdioTransport
import asyncio
from typing import Dict, Any, Optional
import json

class ConfigurationClient:
    """Cliente para gestión de configuración dinámica."""

    def __init__(self, server_path: str):
        self.server_path = server_path
        self._client: Client = None
        self._cache: Dict[str, Any] = {}

    async def conectar(self):
        """Conecta al servidor MCP."""
        server_params = StdioServerParameters(
            command="python",
            args=[self.server_path]
        )
        self._client = Client(StdioTransport(server_params))
        await self._client.connect()

        # Listar recursos disponibles
        resources = await self._client.list_resources()
        print(f"Conectado. {len(resources)} recursos disponibles.")
        for resource in resources:
            print(f"  - {resource.uri}")

    async def obtener_config(
        self,
        seccion: str,
        usar_cache: bool = True
    ) -> Dict[str, Any]:
        """
        Obtiene configuración de una sección.

        Args:
            seccion: Nombre de la sección (database, api, cache, etc.)
            usar_cache: Si usar caché local

        Returns:
            dict con la configuración
        """
        cache_key = f"config://{seccion}"

        # Verificar caché
        if usar_cache and cache_key in self._cache:
            return self._cache[cache_key]

        # Leer recurso
        config = await self._client.read_resource(cache_key)

        # Guardar en caché
        if usar_cache:
            self._cache[cache_key] = config

        return config

    async def obtener_database_config(self) -> Dict[str, Any]:
        """Obtiene configuración de base de datos."""
        return await self.obtener_config("database")

    async def obtener_api_config(self) -> Dict[str, Any]:
        """Obtiene configuración de API."""
        return await self.obtener_config("api")

    async def obtener_cache_config(self) -> Dict[str, Any]:
        """Obtiene configuración de caché."""
        return await self.obtener_config("cache")

    async def obtener_todas_configs(self) -> Dict[str, Dict[str, Any]]:
        """Obtiene todas las configuraciones."""
        secciones = ["database", "api", "cache", "logging", "security"]
        configs = {}

        for seccion in secciones:
            try:
                configs[seccion] = await self.obtener_config(seccion)
            except Exception as e:
                print(f"Error obteniendo config {seccion}: {e}")
                configs[seccion] = None

        return configs

    def invalidar_cache(self, seccion: str = None):
        """
        Invalida la caché de configuración.

        Args:
            seccion: Sección específica o None para invalidar todo
        """
        if seccion is None:
            self._cache.clear()
        else:
            cache_key = f"config://{seccion}"
            self._cache.pop(cache_key, None)

    async def recargar_config(self, seccion: str) -> Dict[str, Any]:
        """
        Fuerza recarga de configuración.

        Args:
            seccion: Sección a recargar

        Returns:
            dict con la nueva configuración
        """
        self.invalidar_cache(seccion)
        return await self.obtener_config(seccion, usar_cache=False)

    async def observar_cambios(
        self,
        seccion: str,
        callback: callable,
        intervalo: int = 60
    ):
        """
        Observa cambios en una configuración y llama al callback.

        Args:
            seccion: Sección a observar
            callback: Función a llamar cuando hay cambios
            intervalo: Intervalo en segundos para verificar
        """
        config_actual = await self.obtener_config(seccion)

        while True:
            await asyncio.sleep(intervalo)

            nueva_config = await self.obtener_config(seccion, usar_cache=False)

            if nueva_config != config_actual:
                await callback(seccion, config_actual, nueva_config)
                config_actual = nueva_config

    async def desconectar(self):
        """Desconecta del servidor."""
        if self._client:
            await self._client.disconnect()

    """Cliente para acceso a catálogos y datos de referencia."""

    def __init__(self, server_path: str):
        self.server_path = server_path
        self._client: Client = None
        self._cache_ttl = 3600  # 1 hora
        self._cache: Dict[str, tuple] = {}  # (data, timestamp)

    async def conectar(self):
        """Conecta al servidor MCP."""
        server_params = StdioServerParameters(
            command="python",
            args=[self.server_path]
        )
        self._client = Client(StdioTransport(server_params))
        await self._client.connect()

    async def _obtener_con_cache(
        self,
        uri: str,
        usar_cache: bool = True
    ) -> Any:
        """Obtiene recurso con caché."""
        import time

        if usar_cache and uri in self._cache:
            data, timestamp = self._cache[uri]
            if time.time() - timestamp < self._cache_ttl:
                return data

        # Leer recurso
        data = await self._client.read_resource(uri)

        # Guardar en caché
        self._cache[uri] = (data, time.time())

        return data

    async def obtener_categorias(self) -> List[Dict[str, Any]]:

        Obtiene lista de categorías.

        Returns:
            list de categorías

        return await self._obtener_con_cache("data://categorias")

        Args:
            categoria_id: ID de la categoría

            dict con la categoría o None

        return await self._obtener_con_cache(

@mcp.tool
def buscar_en_base_datos(
    tabla: Annotated[str, Field(description="Tabla a buscar")],
    filtros: Annotated[Dict[str, Any], Field(description="Filtros de búsqueda")] = None,
    limite: Annotated[int, Field(description="Número máximo de resultados", ge=1, le=100)] = 10
) -> Dict[str, Any]:
    """
    Busca registros en la base de datos.

    Args:
        tabla: Nombre de la tabla (usuarios, productos, ordenes, tickets)
        filtros: Diccionario con filtros {campo: valor}
        limite: Número máximo de resultados

    Returns:
        dict con resultados y metadatos
    """
    if tabla not in DATABASE:
        return {
            "success": False,
            "error": f"Tabla '{tabla}' no existe",
            "tablas_disponibles": list(DATABASE.keys())
        }

    resultados = list(DATABASE[tabla].values())

    # Aplicar filtros
    if filtros:
        resultados_filtrados = []
        for registro in resultados:
            coincide = True
            for campo, valor in filtros.items():
                if campo not in registro or registro[campo] != valor:
                    coincide = False
                    break
            if coincide:
                resultados_filtrados.append(registro)
        resultados = resultados_filtrados

    # Limitar resultados
    resultados = resultados[:limite]

    return {
        "success": True,
        "tabla": tabla,
        "total": len(resultados),
        "resultados": resultados,
        "filtros_aplicados": filtros
    }

class MCPToolWrapper:
    """Wrapper para convertir herramientas MCP en herramientas CrewAI."""

    def __init__(self, client: Client, tool_name: str, description: str):
        self.client = client
        self.tool_name = tool_name
        self.description = description
        self.name = f"mcp_{tool_name}"

    async def execute(self, **kwargs) -> Any:
        """Ejecuta la herramienta MCP."""
        return await self.client.call_tool(self.tool_name, kwargs)

    def to_crewai_tool(self):
        """Convierte a herramienta CrewAI."""

        @tool(self.name)
        def herramienta_wrapper(**kwargs):
            """Herramienta MCP."""
            loop = asyncio.get_event_loop()
            return loop.run_until_complete(self.execute(**kwargs))

        herramienta_wrapper.description = self.description
        return herramienta_wrapper


class CrewAIMCPManager:

class MCPToolRegistry:
    """Registro de herramientas MCP convertidas a CrewAI."""

    def __init__(self, client: Client):
        self.client = client
        self._tools: Dict[str, Any] = {}
        self._crewai_tools: List = []

    async def cargar_herramientas(self):
        """Carga todas las herramientas del servidor MCP."""
        tools = await self.client.list_tools()

        for tool in tools:
            crewai_tool = self._convertir_a_crewai(tool)
            self._tools[tool.name] = crewai_tool
            self._crewai_tools.append(crewai_tool)

        return len(self._tools)

    def _convertir_a_crewai(self, tool_info):
        """Convierte herramienta MCP a CrewAI tool."""
        nombre = tool_info.name
        descripcion = tool_info.description

        @tool(f"mcp_{nombre}")
        async def herramienta_mcp(**kwargs):
            """Ejecuta herramienta MCP."""
            return await self.client.call_tool(nombre, kwargs)

        herramienta_mcp.description = descripcion
        return herramienta_mcp

    def obtener_herramienta(self, nombre: str) -> Any:
        """Obtiene una herramienta específica."""
        return self._tools.get(nombre)

    def obtener_todas(self) -> List:
        """Obtiene todas las herramientas."""
        return self._crewai_tools

    def obtener_por_categoria(self, categoria: str) -> List:
        """Obtiene herramientas por categoría."""
        # Categorías: database, notification, report, query
        categorias = {
            "database": ["buscar_en_base_datos", "crear_registro",
                        "actualizar_registro", "eliminar_registro"],
            "notification": ["enviar_notificacion"],
            "report": ["generar_reporte"],
            "query": ["ejecutar_consulta"]
        }

        herramientas = []
        for nombre in categorias.get(categoria, []):
            if nombre in self._tools:
                herramientas.append(self._tools[nombre])

        return herramientas

class MultiAgentCrewSystem:
    """Sistema completo multi-agente con MCP."""

    def __init__(self, server_path: str):
        self.server_path = server_path
        self.client: Client = None
        self.tool_registry: MCPToolRegistry = None
        self.agent_factory: DynamicAgentFactory = None
        self.task_builder: TaskBuilder = None

    async def inicializar(self):
        """Inicializa el sistema."""
        # Conectar al servidor MCP
        server_params = StdioServerParameters(
            command="python",
            args=[self.server_path]
        )
        self.client = Client(StdioTransport(server_params))
        await self.client.connect()

        # Cargar herramientas
        self.tool_registry = MCPToolRegistry(self.client)
        num_tools = await self.tool_registry.cargar_herramientas()

        # Crear factorías
        self.agent_factory = DynamicAgentFactory(self.tool_registry)
        self.task_builder = TaskBuilder()

        print(f"Sistema inicializado con {num_tools} herramientas")

    async def crear_crew_analisis_completo(
        self,
        datos: str,
        contexto: str = ""
    ) -> Crew:
        """Crea crew para análisis completo."""
        # Agentes
        analista = self.agent_factory.crear_agente(AgentRole.DATA_ANALYST)
        investigador = self.agent_factory.crear_agente(AgentRole.RESEARCHER)
        escritor = self.agent_factory.crear_agente(AgentRole.WRITER)
        coordinador = self.agent_factory.crear_agente(AgentRole.COORDINATOR)

        # Tareas
        tareas = [
            self.task_builder.tarea_analisis_datos(
                analista,
                datos,
                objetivo="identificar patrones y anomalías"
            ),
            self.task_builder.tarea_investigacion(
                investigador,
                f"contexto de los datos: {contexto}",
                profundidad="completa"
            ),
            self.task_builder.tarea_generacion_reporte(
                escritor,
                tipo="analisis",
                filtros={"datos": datos}
            ),
            Task(
                description="Revisa el análisis y el reporte final",
                agent=coordinador,
                expected_output="Reporte validado y aprobado"
            )
        ]

        return Crew(
            agents=[analista, investigador, escritor, coordinador],
            tasks=tareas,
            process=Process.sequential,
            verbose=True
        )

    async def crear_crew_soporte_tecnico(self) -> Crew:
        """Crea crew para soporte técnico."""
        # Agentes
        soporte = self.agent_factory.crear_agente(AgentRole.SUPPORT)
        coordinador = self.agent_factory.crear_agente(AgentRole.COORDINATOR)

        # Tareas (se definirán dinámicamente)
        return Crew(
            agents=[soporte, coordinador],
            tasks=[],  # Se añaden tareas dinámicamente
            process=Process.sequential,
            verbose=True
        )

    async def crear_crew_documentacion(
        self,
        elementos: List[str]
    ) -> Crew:
        """Crea crew para generar documentación."""
        # Agentes
        investigador = self.agent_factory.crear_agente(AgentRole.RESEARCHER)
        escritor = self.agent_factory.crear_agente(AgentRole.WRITER)
        coordinador = self.agent_factory.crear_agente(AgentRole.COORDINATOR)

        # Tareas
        tareas = []
        for elemento in elementos:
            tareas.append(
                self.task_builder.tarea_documentacion(
                    investigador,
                    elemento,
                    tipo="funcion"
                )
            )
            tareas.append(
                Task(
                    description=f"Documenta: {elemento}",
                    agent=escritor,
                    expected_output="Documentación del elemento"
                )
            )

        tareas.append(
            Task(
                description="Compila toda la documentación en un documento final",
                agent=coordinador,
                expected_output="Documento completo con toda la documentación"
            )
        )

        return Crew(
            agents=[investigador, escritor, coordinador],
            tasks=tareas,
            process=Process.sequential,
            verbose=True
        )

    async def ejecutar_analisis(
        self,
        datos: str,
        contexto: str = ""
    ) -> str:
        """Ejecuta análisis completo."""
        crew = await self.crear_crew_analisis_completo(datos, contexto)
        return crew.kickoff()

    async def ejecutar_soporte(
        self,
        ticket_id: str,
        problema: str,
        usuario: str = None
    ) -> str:
        """Ejecuta resolución de soporte."""
        # Crear crew
        crew = await self.crear_crew_soporte_tecnico()

        # Añadir tareas
        soporte = crew.agents[0]
        coordinador = crew.agents[1]

        crew.tasks = [
            self.task_builder.tarea_soporte(soporte, ticket_id, problema, usuario),
            Task(
                description="Valida la solución propuesta",
                agent=coordinador,
                expected_output="Solución validada"
            )
        ]

        return crew.kickoff()

    async def ejecutar_documentacion(
        self,
        elementos: List[str]
    ) -> str:
        """Ejecuta generación de documentación."""
        crew = await self.crear_crew_documentacion(elementos)
        return crew.kickoff()

    async def cerrar(self):
        """Cierra el sistema."""
        if self.client:
            await self.client.disconnect()

Performance

Use caché cuando sea apropiado
# Correcto
from functools import lru_cache
import time

class CachedMCPClient:
    def __init__(self, client: Client, ttl: int = 300):
        self.client = client
        self.ttl = ttl
        self._cache: dict = {}
        self._cache_times: dict = {}

    async def call_tool(self, tool: str, args: dict, use_cache: bool = True):
        if use_cache and tool in self._cache:
            if time.time() - self._cache_times.get(tool, 0) < self.ttl:
                return self._cache[tool]

        result = await self.client.call_tool(tool, args)
        self._cache[tool] = result
        self._cache_times[tool] = time.time()
        return result

asyncio.TimeoutError: Operation timed out

async def call_with_timeout(client, tool, args, timeout=60):
    try:
        result = await asyncio.wait_for(
            client.call_tool(tool, args),
            timeout=timeout
        )
        return result
    except asyncio.TimeoutError:
        print(f"Timeout después de {timeout}s")
        raise
