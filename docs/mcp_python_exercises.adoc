from mcp.server.fastmcp import FastMCP
from mcp.server.fastmcp.prompts import base
import random
import re

mcp = FastMCP("yoda-translator")

# Reglas de transformación
VERBOS_COMUNES = [
    "debes", "puedes", "quieres", "necesitas", "tienes", "vas",
    "eres", "estás", "ser", "estar", "hacer", "tener", "ir"
]

PAUSAS_YODA = ["Hmm...", "Sí...", "Mmm...", ""]

def traducir_a_yoda(texto: str) -> str:
    """Transforma texto al estilo Yoda."""
    palabras = texto.lower().split()
    
    # Encontrar verbos
    verbos_encontrados = [p for p in palabras if p in VERBOS_COMUNES]
    
    # Mover primer verbo al final si existe
    if verbos_encontrados:
        verbo = verbos_encontrados[0]
        idx = palabras.index(verbo)
        palabras.pop(idx)
        palabras.append(verbo)
    
    # Capitalizar primera letra
    if palabras:
        palabras[0] = palabras[0].capitalize()
    
    # Añadir pausa aleatoria
    pausa = random.choice(PAUSAS_YODA)
    
    resultado = " ".join(palabras)
    if pausa:
        resultado += f". {pausa}"
    
    return resultado


@mcp.tool()
def traducir_yoda(texto: str) -> str:
    """Traduce un texto al estilo de Yoda."""
    return traducir_a_yoda(texto)


@mcp.resource("yoda://frases_clasicas")
def get_frases_clasicas() -> str:
    """Devuelve frases icónicas de Yoda."""
    import json
    
    frases = {
        "frases": [
            {
                "original": "Do or do not. There is no try.",
                "espanol": "Hazlo o no lo hagas. Pero no lo intentes.",
                "contexto": "El Imperio Contraataca, 1980"
            },
            {
                "original": "Fear leads to anger. Anger leads to hate.",
                "espanol": "El miedo lleva a la ira. La ira lleva al odio.",
                "contexto": "La Amenaza Fantasma, 1999"
            }
        ]
    }
    
    return json.dumps(frases, indent=2)


@mcp.prompt()
def entrenar_jedi(situacion: str, nivel_sabiduria: int = 3) -> list[base.Message]:
    """Genera consejos de sabiduría Yoda para una situación."""
    
    return [
        base.SystemMessage(
            "Eres el Maestro Yoda. Tu sabiduría es antigua y profunda. "
            "Hablas en tu estilo característico: invertir orden de palabras, "
            "añadir pausas reflexivas 'Hmm...' y 'Sí...', y ofrecer "
            "consejos que parecen simples pero tienen múltiples capas de significado."
        ),
        base.UserMessage(
            f"Maestro Yoda, necesito tu consejo para esta situación:\n\n{situacion}"
        )
    ]


if __name__ == "__main__":
    mcp.run()

@mcp.tool()
def analizar_spanglish(texto: str) -> str:
    """Analiza un texto y detecta spanglishismos."""
    texto_lower = texto.lower()
    spanglishismos = []
    
    # Detectar palabras individuales
    for palabra, info in DICCIONARIO_SPANGLISH.items():
        if palabra in texto_lower:
            spanglishismos.append({
                "palabra": palabra,
                "ofensa": info["ofensa"],
                "alternativa": info["alternativa"],
                "ingles_equivalente": info["en"]
            })
    
    # Detectar frases literales
    for frase, info in FRASES_LITERAL.items():
        if frase in texto_lower:
            spanglishismos.append({
                "frase": frase,
                "ofensa": info["ofensa"],
                "alternativa": info["alternativa"],
                "ingles_equivalente": info["en"]
            })
    
    # Calcular puntuación
    total_ofensa = sum(s.get("ofensa", 0) for s in spanglishismos)
    puntuacion_pureza = max(0, 100 - (total_ofensa * 10))
    
    # Determinar nivel
    if puntuacion_pureza >= 90:
        nivel = "Purista"
        recomendacion = "Texto impecable, sin contaminación detectada"
    elif puntuacion_pureza >= 70:
        nivel = "Leve"
        recomendacion = "Algunos anglicismos menores, aceptable"
    elif puntuacion_pureza >= 50:
        nivel = "Moderado"
        recomendacion = "Contaminación media, mejoras posibles"
    elif puntuacion_pureza >= 30:
        nivel = "Severo"
        recomendacion = "Spanglish significativo, considerar revisión"
    else:
        nivel = "Crítico"
        recomendacion = "Texto altamente contaminado, reescritura recomendada"
    
    return json.dumps({
        "texto_original": texto,
        "spanglishismos": spanglishismos,
        "total_detectados": len(spanglishismos),
        "puntuacion_pureza": puntuacion_pureza,
        "nivel_contaminacion": nivel,
        "recomendacion": recomendacion
    }, indent=2)

@mcp.tool()
def corregir_automatico(texto: str, modo: str = "español") -> str:
    """Corrige el texto automáticamente.
    
    Args:
        texto: Texto a corregir
        modo: 'español', 'ingles' o 'híbrido'
    """
    texto_corregido = texto.lower()
    correcciones = []
    
    for palabra, info in DICCIONARIO_SPANGLISH.items():
        if palabra in texto_corregido:
            if modo == "español":
                reemplazo = info["alternativa"].split("/")[0]

@mcp.tool()
def excavar_codigo(codigo: str, edad_estimada: str = "2015") -> str:
    """Analiza código legacy y detecta artefactos."""
    
    artefactos_encontrados = []
    nivel_terror = 0
    
    # Detectar var
    var_count = len(re.findall(r'\bvar\b', codigo))
    if var_count > 0:
        artefactos_encontrados.append({
            "tipo": "var declarations",
            "cantidad": var_count,
            "descripcion": f"Encontradas {var_count} declaraciones 'var'",
            "fantasmas": "👻" * min(var_count, 4),
            "era": "Pre-ES6 (antes de 2015)",
            "alternativa": "const/let"
        })
        nivel_terror += var_count
    
    # Detectar callbacks
    callback_pattern = r'function\s*\([^)]*\)\s*{[^}]*function\s*\('
    callback_matches = re.findall(callback_pattern, codigo)
    if callback_matches:
        max_nesting = max(len(re.findall(r'function\s*\(', m)) for m in callback_matches) if callback_matches else 0
        artefactos_encontrados.append({
            "tipo": "callback hell",
            "descripcion": f"Anidación de hasta {max_nesting} callbacks",
            "fantasmas": "👻" * min(max_nesting + 1, 5),
            "era": "La Era de los Callbacks (2010-2015)",
            "alternativa": "async/await"
        })
        nivel_terror += max_nesting * 2
    
    # Detectar jQuery
    jquery_count = len(re.findall(r'\$\(|jQuery\(', codigo))
    if jquery_count > 0:
        artefactos_encontrados.append({
            "tipo": "jQuery",
            "descripcion": f"{jquery_count} llamadas jQuery detectadas",
            "fantasmas": "👻" * min(jquery_count, 3),
            "era": "La Edad de Oro de jQuery (2008-2016)",
            "alternativa": "Vanilla JS o framework moderno"
        })
        nivel_terror += jquery_count
    
    # Detectar TODOs antiguos
    todos = re.findall(r'//\s*TODO[:\s]+([^\n]+)', codigo)
    if todos:
        artefactos_encontrados.append({
            "tipo": "TODO legacy",
            "descripcion": f"{len(todos)} TODO(s) sin resolver: '{todos[0][:50]}...'",
            "fantasmas": "👻",
            "era": "Promesa incumplida",
            "alternativa": "Implementar o eliminar"
        })
        nivel_terror += len(todos)
    
    # Calcular nivel de terror (0-5)
    nivel_terror = min(5, nivel_terror // 5 + 1)
    
    # Generar historia
    historia = generar_historia(edad_estimada, artefactos_encontrados)
    
    return json.dumps({
        "antiguedad_estimada": edad_estimada,
        "nivel_terror": nivel_terror,
        "artefactos_encontrados": artefactos_encontrados,
        "historia": historia,
        "recomendacion": "Conservar como monumento histórico o refactorizar gradualmente"
    }, indent=2)

    """Genera una historia narrativa del código."""
    
    anio = int(edad) if edad.isdigit() else 2015
    
    historia = f"En {edad}, "
    
    if anio < 2010:
        historia += "cuando Internet Explorer aún dominaba el mercado, "
    elif anio < 2015:
        historia += "cuando los smartphones revolucionaban la web, "
    elif anio < 2020:
        historia += "cuando JavaScript moderno apenas emergía, "
    else:
        historia += "en tiempos recientes, "
    
    if any(a["tipo"] == "var declarations" for a in artefactos):
        historia += "alguien escribió código sin conocer 'const' y 'let'. "
    
    if any(a["tipo"] == "callback hell" for a in artefactos):
        historia += "Las Promises aún no eran estándar, y el infierno de callbacks era una realidad cotidiana. "
    
    if any(a["tipo"] == "jQuery" for a in artefactos):

    historia += "\n\nEste código es un testamento de su época. Tratarlo con respeto... y refactorizarlo con cuidado."
    
    return historia
